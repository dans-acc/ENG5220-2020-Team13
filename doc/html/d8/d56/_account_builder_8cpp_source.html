258 std::map<int, std::vector<std::pair<std::tm, std::tm>>> AccountBuilder::buildWeekMap(const web::json::value &jsonWeek)
259 {
260  std::map<int, std::vector<std::pair<std::tm, std::tm>>> week;
261  const std::string days[] = JSON_KEY_FENCE_DAYS;
262 
263  // Parse the string representation of time for each week, converting them into the tm (time) structure.
264  std::vector<std::pair<std::tm, std::tm>> dayTimes;
265  std::pair<std::tm, std::tm> dayTime;
266  struct std::tm fromTm = {0};
267  struct std::tm toTm = {0};
268  for (int i = 0; i < days->length(); ++i) {
269 
270  // Check that the week object has a specific day [array].
271  if (!jsonWeek.has_array_field(U(days[i]))) {
272  continue;
273  }
274 
275  // Get the array of times for that day.
276  const web::json::array& times = jsonWeek.at(U(days[i])).as_array();
277  if (times.size() == 0) {
278  continue;
279  }
280 
281  // Parse each of the times, convert the into structures and add.
282  for (auto& time : times) {
283  strptime(time.at(U(JSON_KEY_FENCE_TIME_FROM)).as_string().c_str(), JSON_VALUE_TIME_FORMAT, &fromTm);
284  strptime(time.at(U(JSON_KEY_FENCE_TIME_TO)).as_string().c_str(), JSON_VALUE_TIME_FORMAT, &toTm);
285  dayTime.first = fromTm;
286  dayTime.second = toTm;
287  dayTimes.push_back(dayTime);
288  }
289 
290  // Finally add the vector of times to the map.
291  week.insert(std::pair<int, std::vector<std::pair<std::tm, std::tm>>>(i, dayTimes));
292  dayTimes.clear();
293  }
294 
295  return week;
296 }
