75 void PolyFence::calculateFenceConstants()
76 {
77 
78  // Clear the constants for recalculation.
79  constants.clear();
80  multiples.clear();
81 
82  // If the coordinates are empty, nothing can be done.
83  if (coordinates.empty()) {
84  return;

86 
87  // Calculate poly fence constants.
88  unsigned long i, j = coordinates.size() - 1;
89  for (i = 0; i < coordinates.size(); ++i) {
90  if (coordinates[i].second == coordinates[j].second) {
91  constants.push_back(coordinates[i].first);
92  multiples.push_back(0);
93  } else {
94  constants.push_back(coordinates[i].first

96  / (coordinates[j].second - coordinates[i].second)

98  / (coordinates[j].second - coordinates[i].second));
99  multiples.push_back((coordinates[j].first - coordinates[i].first)
100  / (coordinates[j].second - coordinates[i].second));
