161 Comms* BeeSafeManager::initComms()
162 {
163  auto comms = new Comms();
164 
165  // Try numerous times to initialise the comms interface.
166  bool init = false;
167  short tries = 0;
168  do {
169 
170  // Attempt to initialise the interface.
171  tries++;
172  std::cout << "Comms initialisation attempt " << tries << " / " << INIT_COMMS_TRIES << "..." << std::endl;
173  init = comms->init();
174  if (init) {
175  break;
176  } else {
177  std::cerr << "... comms initialisation attempt " << tries << " / " << INIT_COMMS_TRIES << " failed." << std::endl;
178  }
179 
180  } while (tries < INIT_COMMS_TRIES);
181 
182  // We failed to initialise the interface.
183  if (!init) {
184  delete comms;
185  return nullptr;
186  }
187 
188  // Successfully return the comms instance.
189  return comms;
190 }
